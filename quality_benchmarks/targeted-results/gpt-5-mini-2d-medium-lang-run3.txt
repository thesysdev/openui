root = DeploymentConfig("TaskBoard", [webFrontend, apiServer, worker, database, cache], [ingressWeb], [pvDatabase], [npWeb, npApi, npWorker, npDatabase, npCache], [secretTaskboard])
r_web = Resources(512, 0.5)
p_web = Port(3000, "TCP")
e_web_api = EnvVar("API_URL", null, "api.internal")
e_web_nodeenv = EnvVar("NODE_ENV", "production")
hc_web = HealthCheck("http", "/health", 3000, 30, 5, 3)
webFrontend = ContainerSpec("web-frontend", "taskboard/web:3.2.1", 2, r_web, [p_web], [e_web_api, e_web_nodeenv], hc_web, "public")
r_api = Resources(1024, 1)
p_api = Port(8080, "TCP")
e_api_db = EnvVar("DATABASE_URL", null, "database.internal")
e_api_redis = EnvVar("REDIS_URL", null, "cache.internal")
e_api_jwt = EnvVar("JWT_SECRET", null, null, SecretRef("taskboard-secrets", "jwt-secret"))
hc_api = HealthCheck("http", "/api/health", 8080, 15, 3, 2)
apiServer = ContainerSpec("api-server", "taskboard/api:3.2.1", 3, r_api, [p_api], [e_api_db, e_api_redis, e_api_jwt], hc_api, "internal")
r_worker = Resources(768, 0.5)
e_worker_db = EnvVar("DATABASE_URL", null, "database.internal")
e_worker_redis = EnvVar("REDIS_URL", null, "cache.internal")
e_worker_conc = EnvVar("WORKER_CONCURRENCY", "5")
worker = ContainerSpec("background-worker", "taskboard/worker:3.2.1", 2, r_worker, [], [e_worker_db, e_worker_redis, e_worker_conc], null, "internal")
r_db = Resources(2048, 1)
p_db = Port(5432, "TCP")
e_db_name = EnvVar("POSTGRES_DB", "taskboard")
e_db_user = EnvVar("POSTGRES_USER", "taskboard")
e_db_pass = EnvVar("POSTGRES_PASSWORD", null, null, SecretRef("taskboard-secrets", "db-password"))
database = ContainerSpec("database", "postgres:16", 1, r_db, [p_db], [e_db_name, e_db_user, e_db_pass], null, "internal")
r_cache = Resources(512, 0.25)
p_cache = Port(6379, "TCP")
cache = ContainerSpec("cache", "redis:7.2", 1, r_cache, [p_cache], [], null, "internal")
ingressWeb = IngressRule("taskboard.example.com", "web-frontend", 3000, true)
pvDatabase = PersistentVolume("db-data", "database", 20, "standard", "/var/lib/postgresql/data")
npWeb = NetworkPolicy("np-web", "web-frontend", ["api-server"], [])
npApi = NetworkPolicy("np-api", "api-server", ["web-frontend"], ["database","cache"])
npWorker = NetworkPolicy("np-worker", "background-worker", [], ["database","cache"])
npDatabase = NetworkPolicy("np-database", "database", ["api-server","background-worker"], [])
npCache = NetworkPolicy("np-cache", "cache", ["api-server","background-worker"], [])
secretTaskboard = Secret("taskboard-secrets", ["jwt-secret","db-password"])
