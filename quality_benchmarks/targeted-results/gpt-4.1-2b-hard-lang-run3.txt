root = Workflow([get_projects, get_teams, get_time_entries, get_risks, get_contributors, get_owner_details, get_team_leads, generate_report, send_slack], output_spec)
get_projects = ApiCall("get_projects", "GET", "/api/projects", null, null, null, null, null, "$.projects[?(@.status=='active')]")
get_teams = ApiCall("get_teams", "GET", "/api/teams/{id}", null, null, ["get_projects"], "$.get_projects[*].team_ids", null, "$")
get_time_entries = ApiCall("get_time_entries", "GET", "/api/projects/{id}/time-entries", {"start_date": "2025-10-01", "end_date": "2025-12-31"}, null, ["get_projects"], "$.get_projects[*].id", null, "$.entries")
get_risks = ApiCall("get_risks", "GET", "/api/projects/{id}/risks", null, null, ["get_projects"], "$.get_projects[*].id", null, "$.risks")
get_contributors = ApiCall("get_contributors", "GET", "/api/users/{id}", null, null, ["get_time_entries"], "$.get_time_entries[*].user_id", null, "$")
get_owner_details = ApiCall("get_owner_details", "GET", "/api/users/{id}", null, null, ["get_projects"], "$.get_projects[*].owner_id", null, "$")
get_team_leads = ApiCall("get_team_leads", "GET", "/api/users/{id}", null, null, ["get_teams"], "$.get_teams[*].lead_id", null, "$")
generate_report = ApiCall("generate_report", "POST", "/api/reports/generate", null, {
  "title": "Q4 2025 Project Health Report",
  "sections": "${map(filter(projects, p => p.classification=='at-risk'||p.classification=='critical'), p => ({heading: p.name, content: concat('Classification: ', p.classification, '\\nTotal Hours: ', p.total_hours, '\\nUtilization: ', p.utilization_percent, '%\\nOpen High Risks: ', p.open_high_risks, '\\nBudget Consumed: ', p.budget_consumed_percent, '%\\nOwner: ', p.owner_name, '\\nTeam Leads: ', join(p.team_lead_names, ', '))}))}",
  "recipients": "${dedupe(flatMap(filter(projects, p => p.classification=='at-risk'||p.classification=='critical'), p => [p.owner_email].concat(p.team_lead_emails)))}"
}, ["get_projects", "get_owner_details", "get_team_leads"], null, "$")
send_slack = ApiCall("send_slack", "POST", "/api/notifications/slack", null, {
  "channel": "#project-health",
  "message": "Q4 Project Health Report generated: ${generate_report.url}. ${length(filter(projects, p => p.classification=='at-risk'||p.classification=='critical'))} projects flagged."
}, ["generate_report"], null, "$")
output_spec = WorkflowOutput("{\"report_id\": generate_report.report_id, \"report_url\": generate_report.url, \"flagged_project_count\": length(filter(projects, p => p.classification=='at-risk'||p.classification=='critical')), \"flagged_projects\": map(filter(projects, p => p.classification=='at-risk'||p.classification=='critical'), p => ({\"project_name\": p.name, \"classification\": p.classification, \"owner_name\": p.owner_name, \"utilization_percent\": p.utilization_percent, \"budget_consumed_percent\": p.budget_consumed_percent, \"open_high_risks\": p.open_high_risks}))}")
