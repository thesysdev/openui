root = Pipeline([join_step, date_filter, channel_filter, compute_revenue, compute_cost, compute_profit, compute_margin, agg_category_region, group_filter, agg_category, sort_step], ["category", "regions_active", "combined_revenue", "combined_profit", "best_region", "best_region_revenue"])
join_step = JoinStep("join", "product_id")
date_filter = FilterStep("filter", "order_date", "gte", "2025-07-01")
channel_filter = FilterStep("filter", "channel", "neq", "internal")
compute_revenue = ComputeStep("compute", "revenue", "quantity * unit_price * (1 - discount_percent / 100)")
compute_cost = ComputeStep("compute", "cost", "quantity * unit_cost")
compute_profit = ComputeStep("compute", "profit", "revenue - cost")
compute_margin = ComputeStep("compute", "margin_percent", "(profit / revenue) * 100", [cond_handle_div_zero], "0")
cond_handle_div_zero = Condition("revenue == 0", "0")
agg_category_region = AggregateStep("aggregate", "category,region", [m_order_count, m_total_revenue, m_total_profit, m_avg_margin])
m_order_count = Metric("order_count", "count")
m_total_revenue = Metric("total_revenue", "sum", "revenue", null, 2)
m_total_profit = Metric("total_profit", "sum", "profit", null, 2)
m_avg_margin = Metric("avg_margin", "avg", "margin_percent", null, 1)
group_filter = FilterStep("filter", "order_count", "gte", 2)
agg_category = AggregateStep("aggregate", "category", [m_regions_active, m_combined_revenue, m_combined_profit, m_best_region, m_best_region_revenue])
m_regions_active = Metric("regions_active", "count_distinct", "region")
m_combined_revenue = Metric("combined_revenue", "sum", "total_revenue", null, 2)
m_combined_profit = Metric("combined_profit", "sum", "total_profit", null, 2)
m_best_region = Metric("best_region", "argmax", null, "total_revenue")
m_best_region_revenue = Metric("best_region_revenue", "max", "total_revenue", null, 2)
sort_step = SortStep("sort", "combined_profit", "desc")
