root = Pipeline([f_status, f_hiredate, c_fullname, c_bonus, agg_dept, s_total], ["department", "employee_count", "avg_salary", "total_bonus"])

f_hiredate = FilterStep("filter", "hire_date", "before", "2024-01-01")
c_fullname = ComputeStep("compute", "full_name", "first_name + \" \" + last_name")
cond_bonus_1 = Condition("performance_rating >= 4.5", "salary * 0.15")
cond_bonus_2 = Condition("performance_rating >= 3.5", "salary * 0.10")
cond_bonus_3 = Condition("performance_rating >= 2.5", "salary * 0.05")
cond_bonus_4 = Condition("true", "0")
c_bonus = ComputeStep("compute", "annual_bonus", null, [cond_bonus_1, cond_bonus_2, cond_bonus_3, cond_bonus_4])
m_count = Metric("employee_count", "count")
m_avg_salary = Metric("avg_salary", "avg", "salary", null, 0)
m_total_bonus = Metric("total_bonus", "sum", "annual_bonus", null, 0)
agg_dept = AggregateStep("aggregate", "department", [m_count, m_avg_salary, m_total_bonus])
s_total = SortStep("sort", "total_bonus", "desc")
