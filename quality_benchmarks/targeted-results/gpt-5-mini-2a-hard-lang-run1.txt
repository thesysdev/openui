root = Pipeline([join1, f_date_start, f_not_internal, comp_revenue, comp_cost, comp_profit, comp_margin, agg1, f_group_count, agg2, sort_final], ["category", "regions_active", "combined_revenue", "combined_profit", "best_region", "best_region_revenue"])
join1 = JoinStep("join", "product_id")
f_date_start = FilterStep("filter", "order_date", "gte", "2025-07-01")
f_not_internal = FilterStep("filter", "channel", "neq", "internal")
comp_revenue = ComputeStep("compute", "revenue", "quantity * unit_price * (1 - discount_percent / 100)")
comp_cost = ComputeStep("compute", "cost", "quantity * unit_cost")
comp_profit = ComputeStep("compute", "profit", "revenue - cost")
comp_margin = ComputeStep("compute", "margin_percent", "(profit / (revenue == 0 ? 1 : revenue)) * 100", [cond_nan], "0")
cond_nan = Condition("revenue != 0", "(profit / revenue) * 100")
agg1 = AggregateStep("aggregate", "category,region", [m_order_count, m_total_revenue, m_total_profit, m_avg_margin])
m_order_count = Metric("order_count", "count")
m_total_revenue = Metric("total_revenue", "sum", "revenue", null, 2)
m_total_profit = Metric("total_profit", "sum", "profit", null, 2)
m_avg_margin = Metric("avg_margin", "avg", "margin_percent", null, 1)
f_group_count = FilterStep("filter", "order_count", "gte", 2)
agg2 = AggregateStep("aggregate", "category", [m_regions_active, m_combined_revenue, m_combined_profit, m_best_region_revenue, m_best_region])
m_regions_active = Metric("regions_active", "count_distinct", "region")
m_combined_revenue = Metric("combined_revenue", "sum", "total_revenue", null, 2)
m_combined_profit = Metric("combined_profit", "sum", "total_profit", null, 2)
m_best_region_revenue = Metric("best_region_revenue", "argmax", "total_revenue", "total_revenue", 2)
m_best_region = Metric("best_region", "argmax", "total_revenue", "region")
sort_final = SortStep("sort", "combined_profit", "desc")
