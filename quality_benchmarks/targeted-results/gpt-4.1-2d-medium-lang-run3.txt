root = DeploymentConfig("TaskBoard", [web, api, worker, db, cache], [web_ingress], [db_volume], [web_net, api_net, worker_net, db_net, cache_net], [main_secret])
web = ContainerSpec("web-frontend", "taskboard/web:3.2.1", 2, web_res, [web_port], [web_env_api, web_env_node], web_health, "public")
api = ContainerSpec("api-server", "taskboard/api:3.2.1", 3, api_res, [api_port], [api_env_db, api_env_redis, api_env_jwt], api_health, "internal")
worker = ContainerSpec("background-worker", "taskboard/worker:3.2.1", 2, worker_res, [], [worker_env_db, worker_env_redis, worker_env_conc], null, "none")
db = ContainerSpec("database", "postgres:16", 1, db_res, [db_port], [db_env_db, db_env_user, db_env_pw], null, "internal")
cache = ContainerSpec("cache", "redis:7.2", 1, cache_res, [cache_port], [], null, "internal")
web_res = Resources(512, 0.5)
api_res = Resources(1024, 1)
worker_res = Resources(768, 0.5)
db_res = Resources(2048, 1)
cache_res = Resources(512, 0.25)
web_port = Port(3000, "TCP")
api_port = Port(8080, "TCP")
db_port = Port(5432, "TCP")
cache_port = Port(6379, "TCP")
web_env_api = EnvVar("API_URL", null, "api-server")
web_env_node = EnvVar("NODE_ENV", "production")
api_env_db = EnvVar("DATABASE_URL", null, "database")
api_env_redis = EnvVar("REDIS_URL", null, "cache")
api_env_jwt = EnvVar("JWT_SECRET", null, null, api_jwt_secretref)
api_jwt_secretref = SecretRef("taskboard-secrets", "jwt-secret")
worker_env_db = EnvVar("DATABASE_URL", null, "database")
worker_env_redis = EnvVar("REDIS_URL", null, "cache")
worker_env_conc = EnvVar("WORKER_CONCURRENCY", "5")
db_env_db = EnvVar("POSTGRES_DB", "taskboard")
db_env_user = EnvVar("POSTGRES_USER", "taskboard")
db_env_pw = EnvVar("POSTGRES_PASSWORD", null, null, db_pw_secretref)
db_pw_secretref = SecretRef("taskboard-secrets", "db-password")
web_health = HealthCheck("http", "/health", 3000, 30, 5, 3)
api_health = HealthCheck("http", "/api/health", 8080, 15, 3, 2)
web_ingress = IngressRule("taskboard.example.com", "web-frontend", 3000, true)
db_volume = PersistentVolume("db-data", "database", 20, "standard", "/var/lib/postgresql/data")
web_net = NetworkPolicy("web-frontend-policy", "web-frontend", [], ["api-server"])
api_net = NetworkPolicy("api-server-policy", "api-server", ["web-frontend"], ["database", "cache"])
worker_net = NetworkPolicy("background-worker-policy", "background-worker", [], ["database", "cache"])
db_net = NetworkPolicy("database-policy", "database", ["api-server", "background-worker"], [])
cache_net = NetworkPolicy("cache-policy", "cache", ["api-server", "background-worker"], [])
main_secret = Secret("taskboard-secrets", ["jwt-secret", "db-password"])
